  84     * @param fields e.g. `['field1', {field2:transformer, field3:[sourceName, transformer]}, ...]`

  85     * @param source e.g. `{value: 5, name:'Joe'}`

  86     */

  87    addRequired(fields:Array<string|TransformerMap>, source:StringMap, required=true) {

  88        fields.forEach(f => {

  89            if (typeof f === 'string') {

  90                if (required && source[f] === undefined) {

  91                    log.warn(`required '${f}' not present on ${this.constructor.name}\n${log.inspect(source)}`);

  92                } else {

  93                    this[f] = source[f];

  94                }

  95            } else {

  96                Object.keys(f).forEach(key => {

  97                    const fn = f[key];

  98                    const val = fn instanceof Array? fn[1](fn[0], source) : (<Transformer>fn)(key, source);

  99                    if (required && val === undefined) {

 100                        log.warn(`required '${key}' not present on ${this.constructor.name}\n${log.inspect(source)}`);

 101                    } else {

 102                        this[key] = val;

 103                    }

 104                })

 105            }

 106        })

 107    }

 108

 109    addOptional(fields:Array<string|{[key:string]:Transformer}>, source:any) {

 110        this.addRequired(fields, source, false);

 111    }

 469        const startMS    = new Date(node.attributes.startDate).getTime();

 470        const endMS      = new Date(node.attributes.endDate).getTime();

 475        this.route['endDate']       = endMS - startMS;

 480    add(node:sax.Tag):HkNode {

 481        switch (node.name) {

 482            /**

 483             * <!ELEMENT MetadataEntry EMPTY>

 484             * <!ATTLIST MetadataEntry

 486             *      value CDATA #REQUIRED

 487             * >

 488             */

 489            case 'MetadataEntry':   

 490                this.route[node.attributes.key] = node.attributes.value;

 491                return undefined;

 492            /**

 498            case 'FileReference':

 501            

 502            default:
